	title = modusa.3.t
	desc = modusa.3.d
	picture = GFX_report_event_generic_riot

	mean_time_to_happen = {
		days = 100
		modifier = {
			factor = 1.5
			has_idea = fight_socialism
		}
		modifier = {
			factor = 1.5
			has_country_leader = { ruling_only = yes name = "Thomas Edmund Dewey" }
		}
		modifier = {
			factor = 1.5
			has_idea = socialism_tamed
		}
		modifier = {
			factor = 3
			has_country_leader = { ruling_only = yes name = "Franklin Delano Roosevelt" }
		}
	}
	
	fire_only_once = yes

	hidden = no

	trigger = {
		OR = {
			has_idea = socialist_conflict
			has_idea = fight_socialism
			has_idea = socialism_tamed
		}
		NOT = { has_country_leader = { ruling_only = yes name = "Huey Pierce Long Jr." } }
		democratic < 0.5
		has_country_flag = chicago_massacre
	}

	option = { #This is the United States! Not the Disunited States!
		name = modusa.3.a
		ai_chance = {
			factor = 1
		}
		country_event = { days = 10 id = modusa.4 }
		SSA = {
			add_state_core = 820
			transfer_state = 820
			add_state_core = 358
			transfer_state = 358
			add_state_core = 393
			transfer_state = 393
			set_capital = 820
		}
		add_popularity = {
			ideology = socialism
			popularity = -0.2
		}
		add_popularity = {
			ideology = communism
			popularity = -0.1
		}
	}
}
