country_event = { # England Puppeting Scotland
	id = sco.0
	title = sco.0.t
	desc = sco.0.d
	picture = GFX_report_event_german_troops

	fire_only_once = yes

	is_triggered_only = yes

	option = { # Accept
		name = sco.0.a
		ai_chance = { factor = 100 }
		ENG = {
			if = {
				limit = {
					has_dlc = "Together for Victory"
				}
				set_autonomy = {
					target = SCO
					autonomous_state = autonomy_integrated_puppet
					freedom_level = 0.35
				}
			}
			else = {
				puppet = SCO
			}
		}
	}

	option = { # Reject
		name = sco.0.b
		ai_chance = { factor = 0 }
		IRE = {
			country_event = {
			    id = sco.1
			    days = 3
			    random = 0
			}
		}
		DEN = {
			country_event = {
				id = sco.2
				days = 5
				random = 0
			}
		}
	}
}
