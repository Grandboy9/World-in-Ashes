on_actions = {
	on_startup = {
		effect = {
			every_country = {
				
			set_variable = { reform_points = 10 }
			set_variable = { reform_points_rate = 2 }
			set_variable = { reform_points_max = 500 }
			}
		}
	}
	
	on_daily = {
		effect = {
			add_to_variable = { reform_points = reform_points_rate }
		}
	}
}
